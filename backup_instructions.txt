# Simple GitHub Backup Commands
# Copy và paste từng lệnh sau khi tạo repository trên GitHub

echo "# ContestHub Backup Commands"
echo ""
echo "1. Tạo repository 'ContestHub_backup' trên GitHub"
echo "2. Copy URL repository (VD: https://github.com/yourusername/ContestHub_backup.git)"
echo "3. Chạy các lệnh sau:"
echo ""
echo "# Thêm remote (thay YOUR_GITHUB_URL)"
echo "git remote add origin YOUR_GITHUB_URL"
echo ""
echo "# Push code"
echo "git push -u origin main"
echo ""
echo "# Kiểm tra kết quả"
echo "git remote -v"
echo "git log --oneline"

# Hiển thị thông tin hiện tại
echo ""
